    <script>
  function startVoiceflowChat(portal) {
    // portal: 'buyer' | 'seller' | 'agent'
    if (!window.voiceflow || !window.voiceflow.chat) {
      console.warn('Voiceflow not ready yet, retrying...');
      setTimeout(() => startVoiceflowChat(portal), 700);
      return;
    }

    try {
      // Mümkünse widget oturumunu sıfırla
      if (typeof window.voiceflow.chat.clear === 'function') {
        window.voiceflow.chat.clear();
      } else if (typeof window.voiceflow.chat.reset === 'function') {
        window.voiceflow.chat.reset();
      }
    } catch (err) {
      console.warn('Voiceflow reset error:', err);
    }

    // Sohbet penceresini aç
    window.voiceflow.chat.open();

    // Hangi portaldan geldiysek o intent ile başlat
    if (portal) {
      try {
        window.voiceflow.chat.interact({
          type: 'intent',
          payload: {
            intent: { name: portal },
            entities: []
          }
        });
      } catch (err) {
        console.warn('Voiceflow interact error:', err);
      }
    }
  }

  // HTML'deki onclick'ler için global yap
  window.startVoiceflowChat = startVoiceflowChat;
</script>
